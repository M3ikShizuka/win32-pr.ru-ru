```
propertyDescription
   name = System.Contact.BusinessAddressCity
   shellPKey = PKEY_Contact_BusinessAddressCity
   formatID = 402B5934-EC5A-48C3-93E6-85E86A2D934E

      IsColumn = true
   typeInfo
```
