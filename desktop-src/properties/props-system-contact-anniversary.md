Годовщина контакта.

```

   name = System.Contact.Anniversary
   shellPKey = PKEY_Contact_Anniversary
   formatID = 9AD5BADB-CEA7-4470-A03D-B84E51B9949E
   propID = 100
   SearchInfo
      InInvertedIndex = false

      type = DateTime
```
