```
propertyDescription
   name = System.Devices.AepService.Bluetooth.ServiceGuid
   shellPKey = PKEY_Devices_AepService_Bluetooth_ServiceGuid
   formatID = A399AAC7-C265-474E-B073-FFCE57721716
```
